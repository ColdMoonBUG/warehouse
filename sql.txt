create table wh_account
(
    id            varchar(32)                           not null comment '主键ID'
        primary key,
    username      varchar(50)                           not null comment '用户名',
    display_name  varchar(255)                          not null comment '显示名称',
    role          varchar(20) default 'salesperson'     not null comment '角色: admin/salesperson',
    employee_id   varchar(32)                           null comment '关联业务员ID',
    password_hash varchar(64)                           not null comment '密码hash',
    gesture_hash  varchar(64)                           null comment '手势密码hash',
    status        varchar(10) default 'active'          not null comment '状态: active/inactive',
    created_at    datetime    default CURRENT_TIMESTAMP not null comment '创建时间',
    updated_at    datetime    default CURRENT_TIMESTAMP not null on update CURRENT_TIMESTAMP comment '更新时间',
    constraint uk_username
        unique (username)
)
    comment '账户表' charset = utf8mb4;

create index idx_status
    on wh_account (status);
